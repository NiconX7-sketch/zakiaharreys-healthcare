// Add this to corporate-wellness.html to load programs from database async function loadCorporatePrograms() { const { data } = await supabase.from('corporate_wellness').select('*').eq('active', true); const container = document.getElementById('programs-grid'); if (data?.length) { container.innerHTML = data.map(p => `

${p.title}

${p.description}

$${p.price}
`).join(''); } }
